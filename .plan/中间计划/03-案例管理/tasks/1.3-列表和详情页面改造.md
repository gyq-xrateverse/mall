#### 搜索功能增强

```javascript
// 搜索方法
methods: {
  handleSearchList() {
    this.listQuery.pageNum = 1
    this.getList()
  },

  async getList() {
    this.listLoading = true
    try {
      const response = await this.$http.post('/admin/case/list', this.listQuery)
      if (response.data.code === 200) {
        this.list = response.data.data.list
        this.total = response.data.data.total
      }
    } catch (error) {
      this.$message.error('获取列表失败')
    } finally {
      this.listLoading = false
    }
  },

  async handleUpdateShowStatus(showStatus, row) {
    try {
      await this.$confirm(`确认${showStatus === 1 ? '显示' : '隐藏'}该案例？`, '提示')

      const response = await this.$http.post(`/admin/case/update/showStatus/${row.id}`, {
        showStatus: showStatus
      })

      if (response.data.code === 200) {
        this.$message.success('状态更新成功')
        row.showStatus = showStatus
      } else {
        this.$message.error('状态更新失败')
        // 回滚状态
        row.showStatus = showStatus === 1 ? 0 : 1
      }
    } catch (error) {
      if (error !== 'cancel') {
        this.$message.error('操作失败')
        // 回滚状态
        row.showStatus = showStatus === 1 ? 0 : 1
      }
    }
  }
}
```

#### 详情逻辑处理

```javascript
export default {
  name: 'CaseDataDetail',
  data() {
    return {
      caseData: {},
      relatedCases: [],
      tagList: [],
      isVideoPlaying: false,
      errorMessage: ''
    }
  },

  methods: {
    async getCaseDetail() {
      try {
        const response = await this.$http.get(`/admin/case/${this.$route.params.id}`)
        if (response.data.code === 200) {
          this.caseData = response.data.data

          // 处理标签
          if (this.caseData.tags) {
            this.tagList = this.caseData.tags.split(',').filter(tag => tag.trim())
          }

          // 获取相关案例
          await this.getRelatedCases()
        } else {
          this.errorMessage = response.data.message || '案例不存在'
        }
      } catch (error) {
        console.error('获取案例详情失败:', error)
        this.errorMessage = '获取案例详情失败'
      }
    },

    async getRelatedCases() {
      try {
        const response = await this.$http.get('/admin/case/related', {
          params: {
            categoryId: this.caseData.categoryId,
            excludeId: this.caseData.id,
            limit: 6
          }
        })

        if (response.data.code === 200) {
          this.relatedCases = response.data.data.list
        }
      } catch (error) {
        console.error('获取相关案例失败:', error)
      }
    },

    // 视频播放控制
    handleVideoPlay() {
      this.isVideoPlaying = true
    },

    handleVideoPause() {
      this.isVideoPlaying = false
    },

    handleVideoEnded() {
      this.isVideoPlaying = false
    },

    handleEdit() {
      this.$router.push(`/case/data/update/${this.caseData.id}`)
    },

    handleBack() {
      this.$router.push('/case/data/index')
    },

    buildUrl(objectName) {
      if (!objectName) return ''
      if (objectName.startsWith('http')) return objectName
      return `${process.env.VUE_APP_STORAGE_BASE_URL}/${objectName}`
    },

    formatTime(timeString) {
      return new Date(timeString).toLocaleString()
    }
  },

  async created() {
    await this.getCaseDetail()
  }
}
```
