### 1. 创建分类列表页面
**文件路径：** `mall-admin-web/src/views/case/category/index.vue`

```vue
<template>
  <div class="app-container">
    <!-- 搜索筛选 -->
    <el-card class="filter-container" shadow="never">
      <div>
        <i class="el-icon-search"></i>
        <span>筛选搜索</span>
        <el-button
          style="float:right"
          type="primary"
          @click="handleSearchList()"
          size="small">
          查询搜索
        </el-button>
        <el-button
          style="float:right;margin-right: 15px"
          @click="handleResetSearch()"
          size="small">
          重置
        </el-button>
      </div>
      <div style="margin-top: 15px">
        <el-form :inline="true" :model="listQuery" size="small" label-width="140px">
          <el-form-item label="分类名称：">
            <el-input
              v-model="listQuery.name"
              class="input-width"
              placeholder="分类名称"
              clearable></el-input>
          </el-form-item>
          <el-form-item label="状态：">
            <el-select
              v-model="listQuery.status"
              placeholder="全部"
              clearable
              class="input-width">
              <el-option label="启用" :value="1"></el-option>
              <el-option label="禁用" :value="0"></el-option>
            </el-select>
          </el-form-item>
        </el-form>
      </div>
    </el-card>

    <!-- 操作按钮 -->
    <el-card class="operate-container" shadow="never">
      <i class="el-icon-tickets"></i>
      <span>数据列表</span>
      <el-button
        size="mini"
        class="btn-add"
        @click="handleAdd()"
        style="margin-left: 20px">
        添加
      </el-button>
      <el-button
        size="mini"
        class="btn-add"
        @click="handleDelete()"
        :disabled="multipleSelection.length<1">
        批量删除
      </el-button>
    </el-card>

    <!-- 表格 -->
    <div class="table-container">
      <el-table ref="categoryTable"
                :data="list"
                style="width: 100%;"
                @selection-change="handleSelectionChange"
                v-loading="listLoading"
                border>
        <el-table-column type="selection" width="60" align="center"></el-table-column>
        <el-table-column label="编号" width="100" align="center">
          <template slot-scope="scope">{{ scope.row.id }}</template>
        </el-table-column>
        <el-table-column label="分类名称" align="center">
          <template slot-scope="scope">{{ scope.row.name }}</template>
        </el-table-column>
        <el-table-column label="排序" width="100" align="center">
          <template slot-scope="scope">
            <el-input
              v-model="scope.row.sort"
              @change="handleSortChange(scope.row)"
              size="mini"
              style="width: 60px"></el-input>
          </template>
        </el-table-column>
        <el-table-column label="状态" width="100" align="center">
          <template slot-scope="scope">
            <el-switch
              @change="handleStatusChange(scope.row)"
              :active-value="1"
              :inactive-value="0"
              v-model="scope.row.status">
            </el-switch>
          </template>
        </el-table-column>
        <el-table-column label="显示状态" width="100" align="center">
          <template slot-scope="scope">
            <el-switch
              @change="handleShowStatusChange(scope.row)"
              :active-value="1"
              :inactive-value="0"
              v-model="scope.row.showStatus">
            </el-switch>
          </template>
        </el-table-column>
        <el-table-column label="创建时间" width="180" align="center">
          <template slot-scope="scope">{{ scope.row.createTime | formatDateTime }}</template>
        </el-table-column>
        <el-table-column label="操作" width="200" align="center">
          <template slot-scope="scope">
            <el-button
              size="mini"
              @click="handleUpdate(scope.row)">
              编辑
            </el-button>
            <el-button
              size="mini"
              type="danger"
              @click="handleDelete(scope.row)">
              删除
            </el-button>
          </template>
        </el-table-column>
      </el-table>
    </div>

    <!-- 分页 -->
    <div class="pagination-container">
      <el-pagination
        background
        @size-change="handleSizeChange"
        @current-change="handleCurrentChange"
        layout="total, sizes,prev, pager, next,jumper"
        :current-page.sync="listQuery.pageNum"
        :page-size="listQuery.pageSize"
        :page-sizes="[5,10,15]"
        :total="total">
      </el-pagination>
    </div>
  </div>
</template>

<script>
import {
  fetchCategoryList,
  deleteCategory,
  deleteCategoryBatch,
  updateCategoryStatus,
  updateCategoryShowStatus
} from '@/api/case'
import { formatDate } from '@/utils/date'

const defaultListQuery = {
  pageNum: 1,
  pageSize: 10,
  name: null,
  status: null
}

export default {
  name: 'caseCategoryList',
  data() {
    return {
      listQuery: Object.assign({}, defaultListQuery),
      list: null,
      total: null,
      listLoading: false,
      multipleSelection: []
    }
  },
  filters: {
    formatDateTime(time) {
      if (time == null || time === '') {
        return 'N/A'
      }
      return formatDate(new Date(time), 'yyyy-MM-dd hh:mm:ss')
    }
  },
  created() {
    this.getList()
  },
  methods: {
    handleResetSearch() {
      this.listQuery = Object.assign({}, defaultListQuery)
    },
    handleSearchList() {
      this.listQuery.pageNum = 1
      this.getList()
    },
    handleSizeChange(val) {
      this.listQuery.pageNum = 1
      this.listQuery.pageSize = val
      this.getList()
    },
    handleCurrentChange(val) {
      this.listQuery.pageNum = val
      this.getList()
    },
    handleSelectionChange(val) {
      this.multipleSelection = val
    },
    handleAdd() {
      this.$router.push({ path: '/case/category/add' })
    },
    handleUpdate(row) {
      this.$router.push({ path: '/case/category/update/' + row.id })
    },
    handleDelete(row) {
      if (row) {
        this.$confirm('是否要删除该分类？', '提示', {
          confirmButtonText: '确定',
          cancelButtonText: '取消',
          type: 'warning'
        }).then(() => {
          deleteCategory(row.id).then(response => {
            this.$message({
              message: '删除成功！',
              type: 'success',
              duration: 1000
            })
            this.getList()
          })
        })
      } else {
        if (this.multipleSelection.length < 1) {
          this.$message({
            message: '请选择要删除的分类！',
            type: 'warning',
            duration: 1000
          })
          return
        }
        this.$confirm('是否要删除选中的分类？', '提示', {
          confirmButtonText: '确定',
          cancelButtonText: '取消',
          type: 'warning'
        }).then(() => {
          let ids = []
          for (let i = 0; i < this.multipleSelection.length; i++) {
            ids.push(this.multipleSelection[i].id)
          }
          deleteCategoryBatch(ids).then(response => {
            this.$message({
              message: '删除成功！',
              type: 'success',
              duration: 1000
            })
            this.getList()
          })
        })
      }
    },
    handleStatusChange(row) {
      this.$confirm('是否要修改该状态？', '提示', {
        confirmButtonText: '确定',
        cancelButtonText: '取消',
        type: 'warning'
      }).then(() => {
        updateCategoryStatus(row.id, row.status).then(response => {
          this.$message({
            type: 'success',
            message: '修改成功!'
          })
        })
      }).catch(() => {
        this.$message({
          type: 'info',
          message: '取消修改'
        })
        this.getList()
      })
    },
    handleShowStatusChange(row) {
      this.$confirm('是否要修改该显示状态？', '提示', {
        confirmButtonText: '确定',
        cancelButtonText: '取消',
        type: 'warning'
      }).then(() => {
        updateCategoryShowStatus(row.id, row.showStatus).then(response => {
          this.$message({
            type: 'success',
            message: '修改成功!'
          })
        })
      }).catch(() => {
        this.$message({
          type: 'info',
          message: '取消修改'
        })
        this.getList()
      })
    },
    handleSortChange(row) {
      // 这里可以调用排序API
      this.$message({
        type: 'success',
        message: '排序已更新!'
      })
    },
    getList() {
      this.listLoading = true
      fetchCategoryList(this.listQuery).then(response => {
        this.listLoading = false
        this.list = response.data.list
        this.total = response.data.total
      })
    }
  }
}
</script>
```

```vue
<template>
  <div class="app-container">
    <el-card>
      <div slot="header">
        <span>添加案例分类</span>
        <el-button
          style="float: right; margin: -5px 0"
          type="primary"
          @click="onSubmit()"
          size="small">
          提交
        </el-button>
        <el-button
          style="float: right; margin: -5px 10px 0 0"
          @click="resetForm()"
          size="small">
          重置
        </el-button>
      </div>
      <el-form :model="category"
               :rules="rules"
               ref="categoryFrom"
               label-width="150px"
               size="small">
        <el-form-item label="分类名称：" prop="name">
          <el-input v-model="category.name" class="input-width"></el-input>
        </el-form-item>
        <el-form-item label="排序：">
          <el-input v-model="category.sort" class="input-width" type="number"></el-input>
        </el-form-item>
        <el-form-item label="状态：">
          <el-radio-group v-model="category.status">
            <el-radio :label="1">启用</el-radio>
            <el-radio :label="0">禁用</el-radio>
          </el-radio-group>
        </el-form-item>
        <el-form-item label="显示状态：">
          <el-radio-group v-model="category.showStatus">
            <el-radio :label="1">显示</el-radio>
            <el-radio :label="0">隐藏</el-radio>
          </el-radio-group>
        </el-form-item>
      </el-form>
    </el-card>
  </div>
</template>

<script>
import { createCategory } from '@/api/case'

const defaultCategory = {
  name: '',
  sort: 0,
  status: 1,
  showStatus: 1
}

export default {
  name: 'addCaseCategory',
  data() {
    return {
      category: Object.assign({}, defaultCategory),
      rules: {
        name: [
          { required: true, message: '请输入分类名称', trigger: 'blur' },
          { min: 2, max: 50, message: '长度在 2 到 50 个字符', trigger: 'blur' }
        ]
      }
    }
  },
  methods: {
    onSubmit() {
      this.$refs.categoryFrom.validate((valid) => {
        if (valid) {
          this.$confirm('是否要添加该分类？', '提示', {
            confirmButtonText: '确定',
            cancelButtonText: '取消',
            type: 'warning'
          }).then(() => {
            createCategory(this.category).then(response => {
              this.$refs.categoryFrom.resetFields()
              this.category = Object.assign({}, defaultCategory)
              this.$message({
                message: '添加成功！',
                type: 'success',
                duration: 1000
              })
              this.$router.back()
            })
          })
        } else {
          this.$message({
            message: '验证失败！',
            type: 'error',
            duration: 1000
          })
          return false
        }
      })
    },
    resetForm() {
      this.$refs.categoryFrom.resetFields()
      this.category = Object.assign({}, defaultCategory)
    }
  }
}
</script>

  width: 400px;
}
```

```vue
<script>
import { getCategoryById, updateCategory } from '@/api/case'

const defaultCategory = {
  name: '',
  sort: 0,
  status: 1,
  showStatus: 1
}

export default {
  name: 'updateCaseCategory',
  data() {
    return {
      category: Object.assign({}, defaultCategory),
      originalCategory: Object.assign({}, defaultCategory),
      rules: {
        name: [
          { required: true, message: '请输入分类名称', trigger: 'blur' },
          { min: 2, max: 50, message: '长度在 2 到 50 个字符', trigger: 'blur' }
        ]
      }
    }
  },
  created() {
    this.getCategory()
  },
  methods: {
    onSubmit() {
      this.$refs.categoryFrom.validate((valid) => {
        if (valid) {
          this.$confirm('是否要修改该分类？', '提示', {
            confirmButtonText: '确定',
            cancelButtonText: '取消',
            type: 'warning'
          }).then(() => {
            updateCategory(this.$route.params.id, this.category).then(response => {
              this.$message({
                message: '修改成功！',
                type: 'success',
                duration: 1000
              })
              this.$router.back()

          this.$message({
            message: '验证失败！',

            duration: 1000
          })
          return false
        }
      })
    },

      this.category = Object.assign({}, this.originalCategory)
    },

      getCategoryById(this.$route.params.id).then(response => {
        this.category = response.data

      })
    }
```

### 4. 创建分类选择组件
**文件路径：** `mall-admin-web/src/views/case/components/CategorySelect.vue`

```vue
<template>
  <el-select
    v-model="selectedCategory"
    placeholder="请选择分类"
    clearable
    @change="handleChange">
    <el-option
      v-for="item in categoryOptions"
      :key="item.id"
      :label="item.name"
      :value="item.id">
    </el-option>
  </el-select>
</template>

<script>
import { getAllCategories } from '@/api/case'

export default {
  name: 'CategorySelect',
  props: {
    value: {
      type: [Number, String],
      default: null
    }
  },
  data() {
    return {
      categoryOptions: [],
      selectedCategory: this.value
    }
  },
  watch: {
    value(newVal) {
      this.selectedCategory = newVal
    }
  },
  created() {
    this.getCategoryOptions()
  },
  methods: {
    getCategoryOptions() {
      getAllCategories().then(response => {
        this.categoryOptions = response.data
      })
    },
    handleChange(value) {
      this.$emit('input', value)
      this.$emit('change', value)
    }
  }
}
</script>
```
