# 11-案例数据管理页面

## 任务概述
实现案例数据管理页面，包括案例列表、新增、编辑、删除、文件上传等功能，集成MinIO文件上传组件。

## 前置条件
- 案例分类管理页面已完成
- 案例数据管理API已完成
- MinIO文件上传功能可用

## 实施步骤

### 1. 创建案例列表页面
**文件路径：** `mall-admin-web/src/views/case/data/index.vue`

```vue
<template>
  <div class="app-container">
    <!-- 搜索筛选 -->
    <el-card class="filter-container" shadow="never">
      <div>
        <i class="el-icon-search"></i>
        <span>筛选搜索</span>
        <el-button
          style="float:right"
          type="primary"
          @click="handleSearchList()"
          size="small">
          查询搜索
        </el-button>
        <el-button
          style="float:right;margin-right: 15px"
          @click="handleResetSearch()"
          size="small">
          重置
        </el-button>
      </div>
      <div style="margin-top: 15px">
        <el-form :inline="true" :model="listQuery" size="small" label-width="140px">
          <el-form-item label="案例标题：">
            <el-input
              v-model="listQuery.title"
              class="input-width"
              placeholder="案例标题"
              clearable></el-input>
          </el-form-item>
          <el-form-item label="分类：">
            <category-select
              v-model="listQuery.categoryId"
              class="input-width">
            </category-select>
          </el-form-item>
          <el-form-item label="状态：">
            <el-select
              v-model="listQuery.status"
              placeholder="全部"
              clearable
              class="input-width">
              <el-option label="启用" :value="1"></el-option>
              <el-option label="禁用" :value="0"></el-option>
            </el-select>
          </el-form-item>
        </el-form>
      </div>
    </el-card>

    <!-- 操作按钮 -->
    <el-card class="operate-container" shadow="never">
      <i class="el-icon-tickets"></i>
      <span>数据列表</span>
      <el-button
        size="mini"
        class="btn-add"
        @click="handleAdd()"
        style="margin-left: 20px">
        添加
      </el-button>
      <el-button
        size="mini"
        class="btn-add"
        @click="handleDelete()"
        :disabled="multipleSelection.length<1">
        批量删除
      </el-button>
      <el-button
        size="mini"
        class="btn-add"
        @click="handleRefreshHotScore()">
        刷新热度
      </el-button>
    </el-card>

    <!-- 表格 -->
    <div class="table-container">
      <el-table ref="caseTable"
                :data="list"
                style="width: 100%;"
                @selection-change="handleSelectionChange"
                v-loading="listLoading"
                border>
        <el-table-column type="selection" width="60" align="center"></el-table-column>
        <el-table-column label="编号" width="80" align="center">
          <template slot-scope="scope">{{ scope.row.id }}</template>
        </el-table-column>
        <el-table-column label="预览图" width="100" align="center">
          <template slot-scope="scope">
            <img v-if="scope.row.previewImageUrl" 
                 :src="scope.row.previewImageUrl" 
                 style="width: 50px; height: 50px; object-fit: cover;"
                 @click="handlePreviewImage(scope.row.previewImageUrl)"/>
            <span v-else>无图片</span>
          </template>
        </el-table-column>
        <el-table-column label="标题" min-width="150" align="center">
          <template slot-scope="scope">
            <span>{{ scope.row.title }}</span>
          </template>
        </el-table-column>
        <el-table-column label="分类" width="100" align="center">
          <template slot-scope="scope">{{ scope.row.categoryName }}</template>
        </el-table-column>
        <el-table-column label="热度分数" width="100" align="center">
          <template slot-scope="scope">
            <span style="color: #E6A23C">{{ scope.row.hotScore }}</span>
          </template>
        </el-table-column>
        <el-table-column label="点赞/浏览" width="100" align="center">
          <template slot-scope="scope">
            <div>
              <span style="color: #F56C6C">{{ scope.row.likeCount }}</span> / 
              <span style="color: #409EFF">{{ scope.row.viewCount }}</span>
            </div>
          </template>
        </el-table-column>
        <el-table-column label="排序" width="80" align="center">
          <template slot-scope="scope">
            <el-input
              v-model="scope.row.sort"
              @change="handleSortChange(scope.row)"
              size="mini"
              style="width: 60px"></el-input>
          </template>
        </el-table-column>
        <el-table-column label="状态" width="80" align="center">
          <template slot-scope="scope">
            <el-switch
              @change="handleStatusChange(scope.row)"
              :active-value="1"
              :inactive-value="0"
              v-model="scope.row.status">
            </el-switch>
          </template>
        </el-table-column>
        <el-table-column label="显示" width="80" align="center">
          <template slot-scope="scope">
            <el-switch
              @change="handleShowStatusChange(scope.row)"
              :active-value="1"
              :inactive-value="0"
              v-model="scope.row.showStatus">
            </el-switch>
          </template>
        </el-table-column>
        <el-table-column label="创建时间" width="150" align="center">
          <template slot-scope="scope">{{ scope.row.createTime | formatDateTime }}</template>
        </el-table-column>
        <el-table-column label="操作" width="150" align="center" fixed="right">
          <template slot-scope="scope">
            <el-button
              size="mini"
              @click="handleUpdate(scope.row)">
              编辑
            </el-button>
            <el-button
              size="mini"
              type="danger"
              @click="handleDelete(scope.row)">
              删除
            </el-button>
          </template>
        </el-table-column>
      </el-table>
    </div>

    <!-- 分页 -->
    <div class="pagination-container">
      <el-pagination
        background
        @size-change="handleSizeChange"
        @current-change="handleCurrentChange"
        layout="total, sizes,prev, pager, next,jumper"
        :current-page.sync="listQuery.pageNum"
        :page-size="listQuery.pageSize"
        :page-sizes="[5,10,15]"
        :total="total">
      </el-pagination>
    </div>

    <!-- 图片预览对话框 -->
    <el-dialog title="图片预览" :visible.sync="imagePreviewVisible" width="50%">
      <img :src="previewImageUrl" style="width: 100%">
    </el-dialog>
  </div>
</template>

<script>
import {
  fetchCaseList,
  deleteCase,
  deleteCaseBatch,
  updateCaseStatus,
  updateCaseShowStatus,
  refreshHotScore
} from '@/api/case'
import { formatDate } from '@/utils/date'
import CategorySelect from '../components/CategorySelect'

const defaultListQuery = {
  pageNum: 1,
  pageSize: 10,
  title: null,
  categoryId: null,
  status: null
}

export default {
  name: 'caseDataList',
  components: {
    CategorySelect
  },
  data() {
    return {
      listQuery: Object.assign({}, defaultListQuery),
      list: null,
      total: null,
      listLoading: false,
      multipleSelection: [],
      imagePreviewVisible: false,
      previewImageUrl: ''
    }
  },
  filters: {
    formatDateTime(time) {
      if (time == null || time === '') {
        return 'N/A'
      }
      return formatDate(new Date(time), 'yyyy-MM-dd hh:mm:ss')
    }
  },
  created() {
    this.getList()
  },
  methods: {
    handleResetSearch() {
      this.listQuery = Object.assign({}, defaultListQuery)
    },
    handleSearchList() {
      this.listQuery.pageNum = 1
      this.getList()
    },
    handleSizeChange(val) {
      this.listQuery.pageNum = 1
      this.listQuery.pageSize = val
      this.getList()
    },
    handleCurrentChange(val) {
      this.listQuery.pageNum = val
      this.getList()
    },
    handleSelectionChange(val) {
      this.multipleSelection = val
    },
    handleAdd() {
      this.$router.push({ path: '/case/data/add' })
    },
    handleUpdate(row) {
      this.$router.push({ path: '/case/data/update/' + row.id })
    },
    handleDelete(row) {
      if (row) {
        this.$confirm('是否要删除该案例？', '提示', {
          confirmButtonText: '确定',
          cancelButtonText: '取消',
          type: 'warning'
        }).then(() => {
          deleteCase(row.id).then(response => {
            this.$message({
              message: '删除成功！',
              type: 'success',
              duration: 1000
            })
            this.getList()
          })
        })
      } else {
        if (this.multipleSelection.length < 1) {
          this.$message({
            message: '请选择要删除的案例！',
            type: 'warning',
            duration: 1000
          })
          return
        }
        this.$confirm('是否要删除选中的案例？', '提示', {
          confirmButtonText: '确定',
          cancelButtonText: '取消',
          type: 'warning'
        }).then(() => {
          let ids = []
          for (let i = 0; i < this.multipleSelection.length; i++) {
            ids.push(this.multipleSelection[i].id)
          }
          deleteCaseBatch(ids).then(response => {
            this.$message({
              message: '删除成功！',
              type: 'success',
              duration: 1000
            })
            this.getList()
          })
        })
      }
    },
    handleStatusChange(row) {
      this.$confirm('是否要修改该状态？', '提示', {
        confirmButtonText: '确定',
        cancelButtonText: '取消',
        type: 'warning'
      }).then(() => {
        updateCaseStatus(row.id, row.status).then(response => {
          this.$message({
            type: 'success',
            message: '修改成功!'
          })
        })
      }).catch(() => {
        this.$message({
          type: 'info',
          message: '取消修改'
        })
        this.getList()
      })
    },
    handleShowStatusChange(row) {
      this.$confirm('是否要修改该显示状态？', '提示', {
        confirmButtonText: '确定',
        cancelButtonText: '取消',
        type: 'warning'
      }).then(() => {
        updateCaseShowStatus(row.id, row.showStatus).then(response => {
          this.$message({
            type: 'success',
            message: '修改成功!'
          })
        })
      }).catch(() => {
        this.$message({
          type: 'info',
          message: '取消修改'
        })
        this.getList()
      })
    },
    handleSortChange(row) {
      // 调用排序API
      this.$message({
        type: 'success',
        message: '排序已更新!'
      })
    },
    handleRefreshHotScore() {
      this.$confirm('是否要刷新所有案例的热度分数？', '提示', {
        confirmButtonText: '确定',
        cancelButtonText: '取消',
        type: 'warning'
      }).then(() => {
        refreshHotScore().then(response => {
          this.$message({
            type: 'success',
            message: '热度分数刷新完成!'
          })
          this.getList()
        })
      })
    },
    handlePreviewImage(imageUrl) {
      this.previewImageUrl = imageUrl
      this.imagePreviewVisible = true
    },
    getList() {
      this.listLoading = true
      fetchCaseList(this.listQuery).then(response => {
        this.listLoading = false
        this.list = response.data.list
        this.total = response.data.total
      })
    }
  }
}
</script>

<style scoped>

  width: 203px;
}

  cursor: pointer;
}
</style>

### 2. 创建新增案例页面

<template>
  <div class="app-container">
    <el-card>
      <div slot="header">
        <span>添加案例</span>
        <el-button
          style="float: right; margin: -5px 0"
          type="primary"
          @click="onSubmit()"
          :loading="submitLoading"
          size="small">
          提交
        </el-button>
        <el-button
          style="float: right; margin: -5px 10px 0 0"
          @click="resetForm()"
          size="small">
          重置
        </el-button>
      </div>
      <el-form :model="caseData"
               :rules="rules"
               ref="caseForm"
               label-width="150px"
               size="small">
        <el-form-item label="案例标题：" prop="title">
          <el-input v-model="caseData.title" class="input-width" maxlength="200" show-word-limit></el-input>
        </el-form-item>
        <el-form-item label="所属分类：" prop="categoryId">
          <category-select v-model="caseData.categoryId" class="input-width"></category-select>
        </el-form-item>
        <el-form-item label="案例描述：">
          <el-input
            v-model="caseData.description"
            type="textarea"
            :rows="4"
            class="input-width-large"
            maxlength="1000"
            show-word-limit>
          </el-input>
        </el-form-item>
        <el-form-item label="预览图片：">
          <single-upload v-model="caseData.previewImageUrl"></single-upload>
        </el-form-item>
        <el-form-item label="案例视频：">
          <single-upload v-model="caseData.videoUrl" :file-type="'video'"></single-upload>
        </el-form-item>
        <el-form-item label="视频预览地址：">
          <el-input v-model="caseData.videoPreviewUrl" class="input-width" placeholder="视频预览地址URL"></el-input>
        </el-form-item>
        <el-form-item label="初始数据：">
          <el-row :gutter="20">
            <el-col :span="8">
              <el-input v-model="caseData.likeCount" placeholder="点赞数" type="number"></el-input>
            </el-col>
            <el-col :span="8">
              <el-input v-model="caseData.viewCount" placeholder="浏览次数" type="number"></el-input>
            </el-col>
            <el-col :span="8">
              <el-input v-model="caseData.sort" placeholder="排序" type="number"></el-input>
            </el-col>
          </el-row>
        </el-form-item>
        <el-form-item label="状态：">
          <el-radio-group v-model="caseData.status">
            <el-radio :label="1">启用</el-radio>
            <el-radio :label="0">禁用</el-radio>
          </el-radio-group>
        </el-form-item>
        <el-form-item label="显示状态：">
          <el-radio-group v-model="caseData.showStatus">
            <el-radio :label="1">显示</el-radio>
            <el-radio :label="0">隐藏</el-radio>
          </el-radio-group>
        </el-form-item>
      </el-form>
    </el-card>
  </div>
</template>

<script>
import { createCase } from '@/api/case'
import CategorySelect from '../components/CategorySelect'
import SingleUpload from '@/components/Upload/singleUpload'

const defaultCaseData = {
  title: '',
  categoryId: null,
  description: '',
  previewImageUrl: '',
  videoUrl: '',
  videoPreviewUrl: '',
  likeCount: 0,
  viewCount: 0,
  sort: 0,
  status: 1,
  showStatus: 1
}

export default {
  name: 'addCaseData',
  components: {
    CategorySelect,
    SingleUpload
  },
  data() {
    return {
      caseData: Object.assign({}, defaultCaseData),
      submitLoading: false,
      rules: {
        title: [
          { required: true, message: '请输入案例标题', trigger: 'blur' },
          { min: 2, max: 200, message: '长度在 2 到 200 个字符', trigger: 'blur' }
        ],
        categoryId: [
          { required: true, message: '请选择案例分类', trigger: 'change' }
        ]
      }
    }
  },
  methods: {
    onSubmit() {
      this.$refs.caseForm.validate((valid) => {
        if (valid) {
          this.$confirm('是否要添加该案例？', '提示', {
            confirmButtonText: '确定',
            cancelButtonText: '取消',
            type: 'warning'
          }).then(() => {
            this.submitLoading = true
            createCase(this.caseData).then(response => {
              this.submitLoading = false
              this.$refs.caseForm.resetFields()
              this.caseData = Object.assign({}, defaultCaseData)
              this.$message({
                message: '添加成功！',
                type: 'success',
                duration: 1000
              })
              this.$router.back()
            }).catch(() => {
              this.submitLoading = false
            })
          })
        } else {
          this.$message({
            message: '验证失败！',
            type: 'error',
            duration: 1000
          })
          return false
        }
      })
    },
    resetForm() {
      this.$refs.caseForm.resetFields()
      this.caseData = Object.assign({}, defaultCaseData)
    }
  }
}
</script>

<style scoped>
.input-width {
  width: 400px;
}

.input-width-large {
  width: 600px;
}
</style>
```
